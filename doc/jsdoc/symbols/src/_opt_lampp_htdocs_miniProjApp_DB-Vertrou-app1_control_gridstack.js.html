 19          * Description
 20          * @method is_intercepted
 21          * @param {} a
 22          * @param {} b
 23          * @return UnaryExpression
 24          */
 25         is_intercepted: function (a, b) {
 26             return !(a.x + a.width <= b.x || b.x + b.width <= a.x || a.y + a.height <= b.y || b.y + b.height <= a.y);

 29         /**
 30          * Description
 31          * @method sort
 32          * @param {} nodes
 33          * @param {} dir
 34          * @param {} width
 35          * @return CallExpression
 36          */
 37         sort: function (nodes, dir, width) {
 38             width = width || _.chain(nodes).map(function (node) { return node.x + node.width; }).max().value();
 39             dir = dir != -1 ? 1 : -1;
 40             return _.sortBy(nodes, function (n) { return dir * (n.x + n.y * width); });

146     GridStackEngine.prototype._fix_collisions = function (node) {
147         this._sort_nodes(-1);
148 
149         var nn = node, has_locked = Boolean(_.find(this.nodes, function (n) { return n.locked }));
150         if (!this.float && !has_locked) {
151             nn = {x: 0, y: node.y, width: this.width, height: node.height};
152         }
153 
154         while (true) {
155             var collision_node = _.find(this.nodes, function (n) {
156                 return n != node && Utils.is_intercepted(n, nn);
157             }, this);
158             if (typeof collision_node == 'undefined') {
159                 return;
160             }
161             this.move_node(collision_node, collision_node.x, node.y + node.height,
162                 collision_node.width, collision_node.height, true);
163         }
164     };

177         var collision_node = _.find(this.nodes, function (n) {
178             return Utils.is_intercepted(n, nn);
179         }, this);

187     GridStackEngine.prototype._pack_nodes = function () {
188         this._sort_nodes();
189 
190         if (this.float) {
191             _.each(this.nodes, function (n, i) {
192                 if (n._updating || typeof n._orig_y == 'undefined' || n.y == n._orig_y)
193                     return;
194 
195                 var new_y = n.y;
196                 while (new_y >= n._orig_y) {
197                     var collision_node = _.chain(this.nodes)
198                         .find(function (bn) {
199                             return n != bn && Utils.is_intercepted({x: n.x, y: new_y, width: n.width, height: n.height}, bn);
200                         })
201                         .value();
202 
203                     if (!collision_node) {
204                         n._dirty = true;
205                         n.y = new_y;
206                     }
207                     --new_y;
208                 }
209             }, this);
210         }
211         else {
212             _.each(this.nodes, function (n, i) {
213                 if (n.locked)
214                     return;
215                 while (n.y > 0) {
216                     var new_y = n.y - 1;
217                     var can_be_moved = i == 0;
218 
219                     if (i > 0) {
220                         var collision_node = _.chain(this.nodes)
221                             .take(i)
222                             .find(function (bn) {
223                                 return Utils.is_intercepted({x: n.x, y: new_y, width: n.width, height: n.height}, bn);
224                             })
225                             .value();
226                         can_be_moved = typeof collision_node == 'undefined';
227                     }
228 
229                     if (!can_be_moved) {
230                         break;
231                     }
232                     n._dirty = n.y != new_y;
233                     n.y = new_y;
234                 }
235             }, this);
236         }
237     };

239     GridStackEngine.prototype._prepare_node = function (node, resizing) {
240         node = _.defaults(node || {}, {width: 1, height: 1, x: 0, y: 0 });
241 
242         node.x = parseInt('' + node.x);
243         node.y = parseInt('' + node.y);
244         node.width = parseInt('' + node.width);
245         node.height = parseInt('' + node.height);
246         node.auto_position = node.auto_position || false;
247         node.no_resize = node.no_resize || false;
248         node.no_move = node.no_move || false;
249 
250         if (node.width > this.width) {
251             node.width = this.width;
252         }
253         else if (node.width < 1) {
254             node.width = 1;
255         }
256 
257         if (node.height < 1) {
258             node.height = 1;
259         }
260 
261         if (node.x < 0) {
262             node.x = 0;
263         }
264 
265         if (node.x + node.width > this.width) {
266             if (resizing) {
267                 node.width = this.width - node.x;
268             }
269             else {
270                 node.x = this.width - node.width;
271             }
272         }
273 
274         if (node.y < 0) {
275             node.y = 0;
276         }
277 
278         return node;

308     /**
309      * Description
310      * @method add_node
311      * @param {} node
312      * @return node
313      */
314     GridStackEngine.prototype.add_node = function(node) {
315         node = this._prepare_node(node);
316 
317         if (typeof node.max_width != 'undefined') node.width = Math.min(node.width, node.max_width);
318         if (typeof node.max_height != 'undefined') node.height = Math.min(node.height, node.max_height);
319         if (typeof node.min_width != 'undefined') node.width = Math.max(node.width, node.min_width);
320         if (typeof node.min_height != 'undefined') node.height = Math.max(node.height, node.min_height);
321 
322         node._id = ++id_seq;
323         node._dirty = true;
324 
325         if (node.auto_position) {
326             this._sort_nodes();
327 
328             for (var i = 0; ; ++i) {
329                 var x = i % this.width, y = Math.floor(i / this.width);
330                 if (x + node.width > this.width) {
331                     continue;
332                 }
333                 if (!_.find(this.nodes, function (n) {
334                     return Utils.is_intercepted({x: x, y: y, width: node.width, height: node.height}, n);
335                 })) {
336                     node.x = x;
337                     node.y = y;
338                     break;
339                 }
340             }
341         }
342 
343         this.nodes.push(node);
344 
345         this._fix_collisions(node);
346         this._pack_nodes();
347         this._notify();
348         return node;
349     };

374     GridStackEngine.prototype.can_move_node = function (node, x, y, width, height) {
375         var has_locked = Boolean(_.find(this.nodes, function (n) { return n.locked }));
376 
377         if (!this.height && !has_locked)
378             return true;
379 
380         var cloned_node;
381         var clone = new GridStackEngine(
382             this.width,
383             null,
384             this.float,
385             0,
386             _.map(this.nodes, function (n) { if (n == node) { cloned_node = $.extend({}, n); return cloned_node; } return $.extend({}, n) }));
387 
388         clone.move_node(cloned_node, x, y, width, height);
389 
390         var res = true;
391 
392         if (has_locked)
393             res &= !Boolean(_.find(clone.nodes, function (n) { return n != cloned_node && Boolean(n.locked) && Boolean(n._dirty); }));
394         if (this.height)
395             res &= clone.get_grid_height() <= this.height;
396 
397         return res;
398     };

420     /**
421      * Description
422      * @method move_node
423      * @param {} node
424      * @param {} x
425      * @param {} y
426      * @param {} width
427      * @param {} height
428      * @param {} no_pack
429      * @return node
430      */
431     GridStackEngine.prototype.move_node = function (node, x, y, width, height, no_pack) {
432         if (typeof x != 'number') x = node.x;
433         if (typeof y != 'number') y = node.y;
434         if (typeof width != 'number') width = node.width;
435         if (typeof height != 'number') height = node.height;
436 
437         if (typeof node.max_width != 'undefined') width = Math.min(width, node.max_width);
438         if (typeof node.max_height != 'undefined') height = Math.min(height, node.max_height);
439         if (typeof node.min_width != 'undefined') width = Math.max(width, node.min_width);
440         if (typeof node.min_height != 'undefined') height = Math.max(height, node.min_height);
441 
442         if (node.x == x && node.y == y && node.width == width && node.height == height) {
443             return node;
444         }
445 
446         var resizing = node.width != width;
447         node._dirty = true;
448 
449         node.x = x;
450         node.y = y;
451         node.width = width;
452         node.height = height;
453 
454         node = this._prepare_node(node, resizing);
455 
456         this._fix_collisions(node);
457         if (!no_pack) {
458             this._pack_nodes();
459             this._notify();
460         }
461         return node;
462     };

501     /**
502      * Description
503      * @method GridStack
504      * @param {} el
505      * @param {} opts
506      * @return 
507      */
508     var GridStack = function (el, opts) {
509         var self = this, one_column_mode;
510 
511         this.container = $(el);
512 
513         opts.item_class = opts.item_class || 'grid-stack-item';
514         var is_nested = this.container.closest('.' + opts.item_class).size() > 0;
515 
516         this.opts = _.defaults(opts || {}, {
517             width: parseInt(this.container.attr('data-gs-width')) || 12,
518             height: parseInt(this.container.attr('data-gs-height')) || 0,
519             item_class: 'grid-stack-item',
520             placeholder_class: 'grid-stack-placeholder',
521             handle: '.grid-stack-item-content',
522             cell_height: 60,
523             vertical_margin: 20,
524             auto: true,
525             min_width: 768,
526             float: false,
527             _class: 'grid-stack-' + (Math.random() * 10000).toFixed(0),
528             animate: Boolean(this.container.attr('data-gs-animate')) || false,
529             always_show_resize_handle: opts.always_show_resize_handle || false,
530             resizable: _.defaults(opts.resizable || {}, {
531                 autoHide: !(opts.always_show_resize_handle || false),
532                 handles: 'se'
533             }),
534             draggable: _.defaults(opts.draggable || {}, {
535                 handle: '.grid-stack-item-content',
536                 scroll: false,
537                 appendTo: 'body'
538             })
539         });
540         this.opts.is_nested = is_nested;
541 
542         this.container.addClass(this.opts._class);
543         if (is_nested) {
544             this.container.addClass('grid-stack-nested');
545         }
546 
547         this._init_styles();
548 
549         this.grid = new GridStackEngine(this.opts.width, function (nodes) {
550             var max_height = 0;
551             _.each(nodes, function (n) {
552                 if (n._id == null) {
553                     n.el.remove();
554                 }
555                 else {
556                     n.el
557                         .attr('data-gs-x', n.x)
558                         .attr('data-gs-y', n.y)
559                         .attr('data-gs-width', n.width)
560                         .attr('data-gs-height', n.height);
561                     max_height = Math.max(max_height, n.y + n.height);
562                 }
563             });
564             self._update_styles(max_height + 10);
565         }, this.opts.float, this.opts.height);
566 
567         if (this.opts.auto) {
568             var elements = [];
569             this.container.children('.' + this.opts.item_class).each(function (index, el) {
570                 el = $(el);
571                 elements.push({
572                     el: el,
573                     i: parseInt(el.attr('data-gs-x')) + parseInt(el.attr('data-gs-y')) * parseInt(el.attr('data-gs-width'))
574                 });
575             });
576             _.chain(elements).sortBy(function (x) { return x.i; }).each(function (i) {
577                 self._prepare_element(i.el);
578             }).value();
579         }
580 
581         this.set_animation(this.opts.animate);
582 
583         this.placeholder = $('<div class="' + this.opts.placeholder_class + ' ' + this.opts.item_class + '"><div class="placeholder-content" /></div>').hide();
584         this.container.append(this.placeholder);
585         this.container.height((this.grid.get_grid_height()) * (this.opts.cell_height + this.opts.vertical_margin) - this.opts.vertical_margin);
586 
587         /**
588          * Description
589          * @method on_resize_handler
590          * @return 
591          */
592         var on_resize_handler = function () {
593             if (self._is_one_column_mode()) {
594                 if (one_column_mode)
595                     return;
596 
597                 one_column_mode = true;
598 
599                 self.grid._sort_nodes();
600                 _.each(self.grid.nodes, function (node) {
601                     self.container.append(node.el);
602 
603                     if (!node.no_move) {
604                         node.el.draggable('disable');
605                     }
606                     if (!node.no_resize) {
607                         node.el.resizable('disable');
608                     }
609                 });
610             }
611             else {
612                 if (!one_column_mode)
613                     return;
614 
615                 one_column_mode = false;
616 
617                 _.each(self.grid.nodes, function (node) {
618                     if (!node.no_move) {
619                         node.el.draggable('enable');
620                     }
621                     if (!node.no_resize) {
622                         node.el.resizable('enable');
623                     }
624                 });
625             }
626         };
627 
628         $(window).resize(on_resize_handler);
629         on_resize_handler();
630     };

697     GridStack.prototype._prepare_element = function (el) {
698         var self = this;
699         el = $(el);
700 
701         el.addClass(this.opts.item_class);
702 
703         var node = self.grid.add_node({
704             x: el.attr('data-gs-x'),
705             y: el.attr('data-gs-y'),
706             width: el.attr('data-gs-width'),
707             height: el.attr('data-gs-height'),
708             max_width: el.attr('data-gs-max-width'),
709             min_width: el.attr('data-gs-min-width'),
710             max_height: el.attr('data-gs-max-height'),
711             min_height: el.attr('data-gs-min-height'),
712             auto_position: Utils.toBool(el.attr('data-gs-auto-position')),
713             no_resize: Utils.toBool(el.attr('data-gs-no-resize')),
714             no_move: Utils.toBool(el.attr('data-gs-no-move')),
715             locked: Utils.toBool(el.attr('data-gs-locked')),
716             el: el
717         });
718         el.data('_gridstack_node', node);
719 
720         var cell_width, cell_height;
721 
722         /**
723          * Description
724          * @method on_start_moving
725          * @param {} event
726          * @param {} ui
727          * @return 
728          */
729         var on_start_moving = function (event, ui) {
730             var o = $(this);
731             self.grid.clean_nodes();
732             self.grid.begin_update(node);
733             cell_width = Math.ceil(o.outerWidth() / o.attr('data-gs-width'));
734             cell_height = self.opts.cell_height + self.opts.vertical_margin;
735             self.placeholder
736                 .attr('data-gs-x', o.attr('data-gs-x'))
737                 .attr('data-gs-y', o.attr('data-gs-y'))
738                 .attr('data-gs-width', o.attr('data-gs-width'))
739                 .attr('data-gs-height', o.attr('data-gs-height'))
740                 .show();
741             node.el = self.placeholder;
742         };

744         /**
745          * Description
746          * @method on_end_moving
747          * @param {} event
748          * @param {} ui
749          * @return 
750          */
751         var on_end_moving = function (event, ui) {
752             var o = $(this);
753             node.el = o;
754             self.placeholder.hide();
755             o
756                 .attr('data-gs-x', node.x)
757                 .attr('data-gs-y', node.y)
758                 .attr('data-gs-width', node.width)
759                 .attr('data-gs-height', node.height)
760                 .removeAttr('style');
761             self._update_container_height();
762             self.container.trigger('change', [self.grid.get_dirty_nodes()]);
763 
764             self.grid.end_update();
765         };
766 
767         el.draggable(_.extend(this.opts.draggable, {
768             start: on_start_moving,
769             stop: on_end_moving,
770             /**
771              * Description
772              * @method drag
773              * @param {} event
774              * @param {} ui
775              * @return 
776              */
777             drag: function (event, ui) {
778                 var x = Math.round(ui.position.left / cell_width),
779                     y = Math.floor((ui.position.top + cell_height/2) / cell_height);
780                 if (!self.grid.can_move_node(node, x, y, node.width, node.height)) {
781                     return;
782                 }
783                 self.grid.move_node(node, x, y);
784                 self._update_container_height();
785             },
786             containment: this.opts.is_nested ? this.container.parent() : null
787         })).resizable(_.extend(this.opts.resizable, {
788             start: on_start_moving,
789             stop: on_end_moving,
790             /**
791              * Description
792              * @method resize
793              * @param {} event
794              * @param {} ui
795              * @return 
796              */
797             resize: function (event, ui) {
798                 var x = Math.round(ui.position.left / cell_width),
799                     y = Math.floor((ui.position.top + cell_height/2) / cell_height),
800                     width = Math.round(ui.size.width / cell_width),
801                     height = Math.round(ui.size.height / cell_height);
802                 if (!self.grid.can_move_node(node, x, y, width, height)) {
803                     return;
804                 }
805                 self.grid.move_node(node, x, y, width, height);
806                 self._update_container_height();
807             }
808         }));
809 
810         if (node.no_move || this._is_one_column_mode()) {
811             el.draggable('disable');
812         }
813 
814         if (node.no_resize || this._is_one_column_mode()) {
815             el.resizable('disable');
816         }
817 
818         el.attr('data-gs-locked', node.locked ? 'yes' : null);
819     };

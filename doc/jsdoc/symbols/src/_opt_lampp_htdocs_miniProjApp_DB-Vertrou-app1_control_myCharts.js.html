306   var longitude;
307   var latitude;

309       dataType: "json",
310       url: '../model/getLoc.php',
311       /**
312        * If the ajax success, we launch this method
313        * This function use the JSON file of localisation of container and store this localisation in two variables for the longitude and latitude
314        * @method success
315        * @param {} result
316        */
317       success: function(result){
318         
319         
320         result.forEach(function(d){
321           longitude = d.geolong;
322           latitude = d.geolat;
323         
324         });
325         if(longitude!=null && latitude != null){
326           var var_location = new google.maps.LatLng(latitude,longitude);
327          
328         var var_mapoptions = {
329           center: var_location,
330           zoom: 14
331         };
332  
333     var var_marker = new google.maps.Marker({
334       position: var_location,
335             map: var_map,
336       title:"Venice"});
337  
338         var var_map = new google.maps.Map(document.getElementById("map-container"),
339             var_mapoptions);
340  
341     var_marker.setMap(var_map);

343         }
